236 Date* WeatherDataCollection::parseDate(std::string* dateTimeString) {
237  std::string s = *dateTimeString;
238  size_t space_pos = s.find(' ');
239 
240  // Check if both date and time parts are present
241  if (space_pos == std::string::npos) {
242  return new Date(1, 1, 1900, 0, 0);
243  }
244 
245  std::string date_part = s.substr(0, space_pos); // e.g., "1/01/2010"
246  std::string time_part = s.substr(space_pos + 1); // e.g., "9:00"
247 
248  int day, month, year, hour, minute;
249  char sep1, sep2, sep3;
250 
251  // Parse date (DD/MM/YYYY)
252  std::stringstream date_ss(date_part);
253  if (!(date_ss >> day >> sep1 >> month >> sep2 >> year) || sep1 != '/' || sep2 != '/') {
254  return new Date(1, 1, 1900, 0, 0);
255  }
256 
257  // Parse time (HH:MM)
258  std::stringstream time_ss(time_part);
259  if (!(time_ss >> hour >> sep3 >> minute) || sep3 != ':') {
260  // Time parsing failed, but date is valid, return date with default time 00:00
261  return new Date(day, month, year, 0, 0);
262  }
263 
264  // Success
265  return new Date(day, month, year, hour, minute);
266 }
