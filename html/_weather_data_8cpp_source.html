374  double calculateSPCC(const std::vector<double>& x, const std::vector<double>& y)
375  {
376  if (x.size() != y.size() || x.size() < 2)
377  return 0.0;
378 
379  int n = x.size();
380  double sum_x = 0.0, sum_y = 0.0;
381  double sum_xy = 0.0, sum_x2 = 0.0, sum_y2 = 0.0;
382 
383  for (int i = 0; i < n; ++i) {
384  sum_x += x[i];
385  sum_y += y[i];
386  sum_xy += x[i] * y[i];
387  sum_x2 += x[i] * x[i];
388  sum_y2 += y[i] * y[i];
389  }
390 
391  double numerator = n * sum_xy - sum_x * sum_y;
392  double denominator = std::sqrt((n * sum_x2 - sum_x * sum_x) * (n * sum_y2 - sum_y * sum_y));
393 
394  // Use epsilon for floating point comparison instead of ==
395  if (std::abs(denominator) < 1e-10)
396  return 0.0;
